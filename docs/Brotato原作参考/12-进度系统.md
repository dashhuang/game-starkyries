## 十一、实现建议

### 11.1 进度追踪系统设计

```lua
-- 累计统计（跨局保存）
累计统计 = {
    total_kills = 0,           -- 累计击杀数
    total_materials = 0,       -- 累计收集材料
    total_steps = 0,           -- 累计行走步数
    danger_wins = {0,0,0,0,0,0}, -- 各危险等级通关记录
    character_wins = {},       -- 各角色通关记录（62个）
}

-- 单局统计（每局重置）
单局统计 = {
    current_materials = 0,     -- 当前持有材料
    wave_healing = 0,          -- 本波治疗量
    wave_burn_kills = 0,       -- 本波烧死敌人数
    consumables_collected = 0, -- 本局消耗品数
    weapons_recycled = 0,      -- 本局回收武器数
    turret_count = 0,          -- 当前炮塔数量
    living_trees = 0,          -- 存活树木数量
    unique_weapons = 0,        -- 不同武器种类数
    t4_weapons = 0,            -- T4武器数量
    t1_items = 0,              -- T1道具种类数
}
```
